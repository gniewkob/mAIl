```python
def _connect_and_login(self) -> None:
    self.connection = imaplib.IMAP4_SSL(
        self.mailbox.imap_host,
        self.mailbox.imap_port,
        ssl_context=ssl.create_default_context(),
    )
    try:

                f"IMAP authentication failed for {self.mailbox.imap_user}: {exc}"
            ) from exc
        raise
```

```python
def load_audit_records(path: Path) -> list[dict[str, Any]]:
    if not path.exists():
        return []
    records: list[dict[str, Any]] = []
    with path.open("r", encoding="utf-8") as handle:
        for line in handle:
            line = line.strip()
            if not line:
                continue
            try:
                records.append(json.loads(line))
            except json.JSONDecodeError:
                continue
    return records
```

```python
def test_scrub_state_pii_is_idempotent(tmp_path: Path) -> None:
    """Running scrub_state_pii twice must not corrupt hashes and must return 0 updated_rows on second run."""
    import sqlite3

    db_path = tmp_path / "state.sqlite"
    # Insert a row with PII directly via raw sqlite3 (bypasses acquire_lease hash pre-population)
    conn = sqlite3.connect(db_path)
    conn.execute(
        """
        CREATE TABLE email_processing_state (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            mailbox_id TEXT NOT NULL DEFAULT 'default',
            message_id TEXT,
            fingerprint TEXT NOT NULL,
            sender TEXT,
            sender_sha256 TEXT,
            subject TEXT,
            subject_sha256 TEXT,
            status TEXT NOT NULL,
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL
        )
        """
    )
    conn.execute(
        "INSERT INTO email_processing_state (fingerprint, sender, subject, status, created_at, updated_at) "
        "VALUES (?, ?, ?, ?, datetime('now'), datetime('now'))",
        ("fp-idempotency", "sender@example.com", "Test subject", "processed"),
    )
    conn.commit()
    conn.close()

    # First run — should scrub the row
    result1 = scrub_state_pii(db_path)
    assert result1.updated_rows == 1

    # Verify hashes were computed
    conn = sqlite3.connect(db_path)
    row = conn.execute("SELECT sender, sender_sha256, subject_sha256 FROM email_processing_state WHERE fingerprint = 'fp-idempotency'").fetchone()
    conn.close()
    assert row[0] == "[redacted]"
    assert row[1] is not None and len(row[1]) == 64
    assert row[2] is not None and len(row[2]) == 64
    first_run_sender_sha256 = row[1]

    # Second run — must report 0 updated rows and must NOT corrupt hashes
    result2 = scrub_state_pii(db_path)
    assert result2.updated_rows == 0

    conn = sqlite3.connect(db_path)
    row2 = conn.execute("SELECT sender_sha256 FROM email_processing_state WHERE fingerprint = 'fp-idempotency'").fetchone()
    conn.close()
    assert row2[0] == first_run_sender_sha256, "Hash must be unchanged on second run"
```

### QUAL-06: Fix `_get` helper to preserve falsy values that are explicitly set

The `_get(key, default)` helper in `_normalize_mailbox` uses `value if value is not None else default`. The inconsistency is that `imap_allow_folder_expunge` is handled directly with `raw_mailbox.get(key, default)`, while other fields use `_get`. This is actually correct for `imap_allow_folder_expunge` (handles `False` correctly). The real risk is for `imap_fetch_limit=0` — check whether `_get` handles it. `0 is not None` is `True`, so `_get("imap_fetch_limit", 100)` for a manifest entry `"imap_fetch_limit": 0` correctly returns `0`. The current code is fine for all non-None falsy values.

The actual gap: `_get` falls through to default for `None`, but JSON manifests can explicitly set a field to `null` (Python `None`). If a user wants to explicitly clear a field to `None`, they can't — it falls through to the Settings default. This is an edge case worth documenting.

- [ ] **Step 1: Add a comment documenting the `_get` behavior**

In `src/mail_ai_agent/config.py`, add a comment to `_get`:

```python
def _get(key: str, default: Any) -> Any:
    # Returns the manifest value when present and non-None.
    # Explicitly setting a key to null in JSON will fall through to the
    # Settings default — this is intentional (null means "inherit").
    value = raw_mailbox.get(key)
    return value if value is not None else default
```
